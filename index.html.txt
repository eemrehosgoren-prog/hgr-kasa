// XP KAZAN
function xpKazan(kazanilanXP){
    if(seviye>=10) return;
    xp += kazanilanXP;
    while(seviye < 10 && xp >= seviyeXP[seviye-1]){
        xp -= seviyeXP[seviye-1];
        seviye++;
        document.getElementById("sonuc").innerText += " 🎉 Seviye Atladın!";
        altin += 25; guncelleAltin();
    }
    if(seviye >= 10){ xp = seviyeXP[seviyeXP.length-1]; }
    guncelleXP(); guncelleSeviye();
}

// Günlük Slot Sistemi
function gunlukSlot(){
    let lastPlay = localStorage.getItem("gunlukSlot")? parseInt(localStorage.getItem("gunlukSlot")):0;
    let now = Date.now();
    if(now - lastPlay < 86400000){
        let kalan = Math.ceil((86400000-(now-lastPlay))/3600000);
        document.getElementById("gunluk-sonuc").innerText = "⏱️ Zaten çevirdin! " + kalan + " saat sonra tekrar açabilirsin.";
        return;
    }
    localStorage.setItem("gunlukSlot", now);
    let sans = Math.random()*100, kazan=0;
    if(sans<50) kazan=10; else if(sans<80) kazan=25; else if(sans<95) kazan=50; else kazan=100;
    altin+=kazan; guncelleAltin();
    const floatDiv=document.createElement("div"); floatDiv.className="float-anim"; floatDiv.innerText="+"+kazan+" 💰"; floatDiv.style.left=(window.innerWidth/2-50)+"px"; floatDiv.style.top="150px"; document.body.appendChild(floatDiv); setTimeout(()=>{ floatDiv.remove(); },1000);
    document.getElementById("gunluk-sonuc").innerText = "🎉 Günlük Slot: "+kazan+" Altın kazandın!";
}
